			for (int i = 0; i < dizi.Length-1; i++)
			{
				int min = i;
				for (int j = i; j < dizi.Length; j++)
				{
					if (dizi[j] < dizi[min])
						min = j;
				}
				int a = dizi[min];
				dizi[min] = dizi[i];
				dizi[i] = a;
			}
			foreach (var item in dizi)
				Console.WriteLine(item);
